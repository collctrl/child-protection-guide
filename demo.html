<body>

  <h2>📝 通報建議 Demo</h2>

  

  <label for="caseInput">請輸入案例描述：</label><br>

  <textarea id="caseInput" rows="4" cols="40" placeholder="例如：學生上課時突然哭泣，不願意回應老師..."></textarea>

  <br><br>

  

  <button id="generateButton">✨ 生成通報建議</button>

  <span id="generateSpinner" class="hidden">⏳</span>


  <div id="summaryOutput" class="hidden">

    <h3>生成結果：</h3>

    <p id="summaryText"></p>

    <button id="ttsButton" class="hidden">🔊 朗讀</button>

  </div>


  <script>

  const generateButton = document.getElementById('generateButton');

  const generateSpinner = document.getElementById('generateSpinner');

  const summaryOutput = document.getElementById('summaryOutput');

  const summaryText = document.getElementById('summaryText');

  const ttsButton = document.getElementById('ttsButton');

  const caseInput = document.getElementById('caseInput');


  generateButton.addEventListener('click', async () => {

      const userInput = caseInput.value.trim();

      generateButton.textContent = "生成中...";

      generateSpinner.classList.remove('hidden');


      // 模擬延遲

      await new Promise(resolve => setTimeout(resolve, 1500));


      // 假裝 AI 回覆

      const fakeResponses = [

          "建議先協助學生釐清情緒來源，再引導表達需求。",

          "可以運用深呼吸或短暫休息，幫助學生降低緊張感。",

          "適時引介導師或同儕支持，避免學生感到孤立。",

          "強化正向回饋，提升學生的自我效能感。"

      ];

      let randomText = fakeResponses[Math.floor(Math.random() * fakeResponses.length)];


      // 如果有輸入內容，就讓回覆稍微呼應

      if (userInput) {

          randomText = `針對「${userInput}」，${randomText}`;

      }


      summaryText.textContent = randomText;

      summaryOutput.classList.remove('hidden');

      ttsButton.classList.remove('hidden');


      // 還原按鈕

      generateButton.textContent = "✨ 生成通報建議";

      generateSpinner.classList.add('hidden');

  });


  // 簡單的文字轉語音

  ttsButton.addEventListener('click', () => {

      const utterance = new SpeechSynthesisUtterance(summaryText.textContent);

      utterance.lang = "zh-TW";

      speechSynthesis.speak(utterance);

  });

  </script>
